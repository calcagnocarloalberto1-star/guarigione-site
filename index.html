      const html=document.documentElement,toggle=document.querySelector('[data-theme-toggle]');

      let theme=window.matchMedia('(prefers-color-scheme: dark)').matches?'dark':'light';

      html.setAttribute('data-theme',theme);

      function syncIcon(){

        toggle.innerHTML=theme==='dark'

          ? '<svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><circle cx="12" cy="12" r="5"/><path d="M12 1v2M12 21v2M4.22 4.22l1.42 1.42M18.36 18.36l1.42 1.42M1 12h2M21 12h2M4.22 19.78l1.42-1.42M18.36 5.64l1.42-1.42"/></svg>'

          : '<svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M21 12.79A9 9 0 1 1 11.21 3 7 7 0 0 0 21 12.79z"/></svg>';

        toggle.setAttribute('aria-label',theme==='dark'?'Attiva tema chiaro':'Attiva tema scuro');

      }

      syncIcon();

      toggle.addEventListener('click',function(){theme=theme==='dark'?'light':'dark';html.setAttribute('data-theme',theme);syncIcon()});

      document.getElementById('year').textContent=new Date().getFullYear();

      const header=document.getElementById('header');

      window.addEventListener('scroll',()=>header.classList.toggle('scrolled',window.scrollY>24));

      const mobileBtn=document.getElementById('mobileMenuBtn');

      mobileBtn?.addEventListener('click',()=>{const open=header.classList.toggle('open');mobileBtn.setAttribute('aria-expanded',open?'true':'false')});

      document.querySelectorAll('.nav-panel a').forEach(a=>a.addEventListener('click',()=>header.classList.remove('open')));

    })();

  </script>

</body>

</html>
